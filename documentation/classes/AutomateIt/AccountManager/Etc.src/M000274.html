  def groups_for_user(query)
    pwent = users[query]
    return [] if preview? and not pwent
    username = pwent.name
    result = Set.new
    result << groups[pwent.gid].name if groups[pwent.gid]
    ::Etc.group do |grent|
      result << grent.name if grent.mem.include?(username)
    end
    return result.to_a
  end
