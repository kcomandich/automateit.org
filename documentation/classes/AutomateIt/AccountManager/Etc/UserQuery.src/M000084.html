    def [](query)
      ::Etc.endpwent
      begin
        case query
        when String
          return ::Etc.getpwnam(query)
        when Fixnum
          return ::Etc.getpwuid(query)
        else
          raise TypeError.new("unknonwn type for query: #{query.class}")
        end
      rescue ArgumentError
        return nil
      end
    end
