  def add_groups_to_user(groups, username)
    return _add_groups_to_user_helper(groups, username) do |missing, username|
      targets = (groups_for_user(username) + missing).uniq

      cmd = "usermod -G #{targets.join(',')} #{username}"
      interpreter.sh(cmd)
    end
  end
