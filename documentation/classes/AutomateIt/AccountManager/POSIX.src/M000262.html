  def remove_user(username, opts={})
    return _remove_user_helper(username, opts) do |username, opts|
      # Options: -r -- remove the home directory and mail spool
      cmd = "userdel"
      cmd << " -r" unless opts[:remove_home] == false
      cmd << " #{username}"
      cmd << " > /dev/null" if opts[:quiet]
      interpreter.sh(cmd)
    end
  end
