  def _read
    @contents = \
      if writing? or (preview? and @filename and _exists? and _readable?)
        File.read(@filename)
      elsif preview? and not _readable?
        log.info(PNOTE+"Not allowed to read file, previewing edits as if it doesn't exist: #{@filename}")
        nil
      else
        nil
      end
  end
