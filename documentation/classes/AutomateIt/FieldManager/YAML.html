  def setup(opts={})
    if filename = opts.delete(:file)
      contents = _read(filename)
      binder = interpreter.send(:binding)
      output = HelpfulERB.new(contents, filename).result(binder)

      opts[:struct] = ::YAML::load(output)
    end
    super(opts)
  end
