  def install(*packages)
    return _install_helper(*packages) do |list, opts|
      # easy_install options:
      # -Z : install into a direcory rather than a file
      cmd = "easy_install -Z "+list.join(" ")+" < /dev/null"
      cmd << " > /dev/null" if opts[:quiet]
      cmd << " 2>&1"

      interpreter.sh(cmd)
    end
  end
