  def uninstall(*packages)
    return _uninstall_helper(*packages) do |list, opts|

      cmd = "pecl uninstall "+list.join(" ")+" < /dev/null"
      cmd << " > /dev/null" if opts[:quiet]
      cmd << " 2>&1"

      interpreter.sh(cmd)
    end
  end
