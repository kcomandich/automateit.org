  def setup(opts={})
    super(opts)

    @@struct_cache ||= {}

    if opts[:struct]
      @struct = opts[:struct]
    else
      @struct ||= {}
    end

    # Generate bi-directional map
    @key_aliases ||= @@key_aliases ||= {
      :version => :release,
    }.inject({}){|s,v| s[v[0]] = v[1]; s[v[1]] = v[0]; s}
  end
