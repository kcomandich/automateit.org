  def which!(command)
    result = which(command)
    if result.nil?
      raise ArgumentError.new("command not found: #{command}")
    else
      true
    end
  end

  def _which_helper(&block)
    data = block.call
    data.strip! if data
    return (! data.blank? && File.exists?(data)) ? data : nil
  end
