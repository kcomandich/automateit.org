  def setup(opts={})
    super(opts)
    @struct ||= {}
    @tags   ||= Set.new
    @struct_updated = false
    @our_hostname_tags ||= Set.new
    @our_platform_tags ||= Set.new

    if opts[:struct]
      @struct_updated = true
      @struct.merge!(AutomateIt::TagManager::TagParser.expand(opts[:struct]))
    end
  end
