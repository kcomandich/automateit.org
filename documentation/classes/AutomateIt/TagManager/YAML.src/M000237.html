  def setup(opts={})
    if filename = opts.delete(:file)
      contents = _read(filename)
      output = HelpfulERB.new(contents, filename).result

      text = AutomateIt::TagManager::TagParser.normalize(output)
      opts[:struct] = ::YAML::load(text)
    end
    super(opts)
  end
